# 시계열 기반
## 이동 평균을 통한 날짜별 추이 확인
```sql
  with
  date_range as (
  -- 1. 데이터의 시작일과 종료일을 자동으로 파악하여 빈 날짜 없는 뼈대 생성
  select
  	cast(generate_series(
  	min(cast(dt as date)),
  	max(cast(dt as date)),
  	interval '1 day') as date) as dt
  from purchase_log),
  
  daily_sum as (
  --2. 생성된 날짜 뼈대에 실제 매출 연결 (매출 없는 날은 0으로 처리)
  select
  	r.dt,
  	coalesce(sum(p.purchase_amount),0) as total_amount
  from date_range as r
  left join purchase_log as p
  on r.dt=cast(p.dt as date)
  group by r.dt)
  
  select
  dt,
  total_amount,
  avg(total_amount) over (order by dt rows between 6 preceding and current row) as seven_day_avg_amount,
  case 
  when count(*) over (order by dt
  				   rows between 6 preceding and current row)=7
				   then avg(total_amount) over (order by dt rows between 6 preceding and current row)
				   end as seven_day_avg_amount_strict
    from daily_sum
    order by dt
---------------------------------------------------
SELECT 
    dt,
    total_amount,
    AVG(total_amount) OVER w AS seven_day_avg_amount,
    CASE 
        WHEN COUNT(*) OVER w = 7 THEN AVG(total_amount) OVER w 
    END AS seven_day_avg_amount_strict
FROM daily_sum
WINDOW w AS (ORDER BY dt ROWS BETWEEN 6 PRECEDING AND CURRENT ROW)
ORDER BY dt;
```
## 당월 매출 누계(날짜별 매출&해당 월의 매출 누적액)
```sql
  with
  daily_purchase as (
  --'month' 단위로 절삭하여 해당 월의 1일자로 통일
  select
  	cast(dt as date) as dt,
  	date_trunc('month',cast(dt as date)) as year_month,
  	sum(purchase_amount) as total_amount
  from purchase_log
  group by dt)
  
  select
  dt,
  --TO_CHAR을 통해 원하는 형식(YYYY-MM)으로 가독성 있게 출력
  to_char(year_month,'YYYY-MM') as year_month,
  total_amount,
  sum(total_amount) over (
  partition by year_month order by dt
  rows between unbounded preceding and current row) as agg_amount
  from daily_purchase
  order by dt
```
* 매출 없는 날이 포함된 경우
```sql
  with
  date_series as (
  select
  	cast(generate_series(
  	min(cast(dt as date)),
  	max(cast(dt as date)),
  	interval '1 day') as date) as dt
  from purchase_log),
  
  daily_sum as (
  select
  	s.dt,
  	cast(date_trunc('month',s.dt) as date) as m_dt,
  	coalesce(sum(p.purchase_amount),0) as daily_amt
  from date_series as s
  left join purchase_log as p 
  on s.dt=cast(p.dt as date)
  group by s.dt,m_dt)
  
  select
  dt,
  to_char(m_dt,'YYYY-MM') as month,
  daily_amt,
  sum(daily_amt) over (partition by m_dt order by dt) as monthly_cum_amt
  from daily_sum
```
## 월별 매출의 작대비 구하기
```sql
  with
  monthly_sales as (
  select
  	date_trunc('month',cast(dt as date)) as month,
  	sum(purchase_amount) as total_amount
  from purchase_log
  group by 1),
  
  yearly_comparison as (
  select
  	month,
  	total_amount as current_amount,
  	lag(total_amount,12) over (order by month) as last_year_amount
  from monthly_sales)
  
  select
  to_char(month,'YYYY-MM') as month,
  current_amount,
  last_year_amount,
  current_amount-last_year_amount as diff_amount,
  concat(
  round(cast((current_amount-last_year_amount) as numeric)/
  nullif(last_year_amount,0)*100.0,2),'%') as yoy_growth_rate 
  from yearly_comparison
  order by month desc
```
## Z차트
```sql
  with
  monthly_sales as (
  select
  	date_trunc('month',cast(dt as date)) as year_month,
  	sum(purchase_amount) as monthly_amount
  from purchase_log
  group by 1
  order by 1),
  
  z_chart as (
  select
  	year_month,
  	monthly_amount,
  	sum(monthly_amount) over (
  	partition by extract(year from year_month)
  	order by year_month
  	rows between unbounded preceding and current row) as ytd_amount,
  	avg(monthly_amount) over (
  	order by year_month
  	rows between 11 preceding and current row) as rolling_12m_avg
  from monthly_sales)
  
  select
  to_char(year_month,'YYYY-MM') as year_month,
  monthly_amount,
  ytd_amount,
  rolling_12m_avg
  from z_chart
  where year_month>=date '2015-01-01'
  and year_month<date '2016-01-01'
  order by year_month
```
## 매출을 파악할때 중요 포인트
```sql
    with
  monthly_base as (
  select
  	date_trunc('month',dt::date)::date as year_month,
  	sum(purchase_amount) as monthly_amount,
  	count(distinct order_id) as orders
  from purchase_log
  group by 1
  order by 1),
  
  z_chart as (
  select
  	year_month,
  	monthly_amount,
  	orders,
  	monthly_amount/nullif(orders,0) as aov,
  	sum(monthly_amount) over (
  	partition by extract(year from year_month)
  	order by year_month
  	rows between unbounded preceding and current row) as ytd_amount,
  	lag(monthly_amount,12) over (order by year_month) as last_year_amount
  from monthly_base)
  
  select
  to_char(year_month,'YYYY-MM') as year_month,
  monthly_amount,
  orders,
  aov,
  ytd_amount,
  last_year_amount,
  round(100.0*monthly_amount/nullif(last_year_amount,0),2)
  as yoy_rate
  from z_chart
  order by year_month
```
# 다면적인 축을 통한 매출 집계
## 카테고리별 매출과 소계 계산하기
```sql
	SELECT
	    CASE WHEN GROUPING(category) = 1 THEN 'Total' ELSE category END AS category,
	    CASE WHEN GROUPING(sub_category) = 1 THEN 'Sub-Total' ELSE sub_category END AS sub_category,
	    SUM(price) AS amount,
	    -- 정렬용 가상 컬럼 (합계 행을 맨 아래로 보내기 위함)
	    GROUPING(category) AS grp_cat,
	    GROUPING(sub_category) AS grp_sub
	FROM purchase_detail_log
	GROUP BY ROLLUP(category, sub_category)
	ORDER BY grp_cat, category, grp_sub, sub_category;
```
## ABC 분석
```sql
	WITH category_sales AS (
	    SELECT
	        category,
	        SUM(price) AS amount
	    FROM purchase_detail_log
	    GROUP BY category
	),
	abc_calculation AS (
	    SELECT
	        category,
	        amount,
	        -- 1. 매출 비중 (%)
	        ROUND(100.0 * amount / SUM(amount) OVER(), 2) AS composition_ratio,
	        -- 2. 누적 매출 비중 (%) : ROWS BETWEEN 생략 가능 (ORDER BY 기본값이 UNBOUNDED PRECEDING~CURRENT)
	        ROUND(100.0 * SUM(amount) OVER(ORDER BY amount DESC) / SUM(amount) OVER(), 2) AS cumulative_ratio,
	        -- 3. 매출 순위 추가
	        ROW_NUMBER() OVER(ORDER BY amount DESC) AS rank
	    FROM category_sales
	)
	SELECT
	    rank,
	    category,
	    amount,
	    composition_ratio,
	    cumulative_ratio,
	    -- 경계값 처리를 위해 부등호(<=)를 사용하는 것이 더 안전합니다.
	    CASE
	        WHEN cumulative_ratio <= 70 THEN 'A'
	        WHEN cumulative_ratio <= 90 THEN 'B'
	        ELSE 'C'
	    END AS abc_rank
	FROM abc_calculation
	ORDER BY rank;
```
## 팬 차트로 상품의 매출 증가율 확인하기
```sql
	WITH monthly_category_amount AS (
	    SELECT
	        DATE_TRUNC('month', dt::date)::date AS m_dt, -- 날짜 연산을 위해 DATE 타입 유지
	        category,
	        SUM(price) AS amount
	    FROM purchase_detail_log
	    GROUP BY 1, 2
	),
	fan_chart_base AS (
	    SELECT
	        m_dt,
	        category,
	        amount,
	        -- 카테고리별로 정렬된 첫 번째 매출액을 가져옴
	        FIRST_VALUE(amount) OVER (
	            PARTITION BY category 
	            ORDER BY m_dt
	        ) AS base_amount
	    FROM monthly_category_amount
	)
	SELECT
	    TO_CHAR(m_dt, 'YYYY-MM') AS year_month,
	    category,
	    amount,
	    base_amount,
	    -- 0으로 나누기 방지(NULLIF) 추가 및 반올림
	    ROUND(100.0 * amount / NULLIF(base_amount, 0), 2) AS rate
	FROM fan_chart_base
	ORDER BY year_month, category;
```

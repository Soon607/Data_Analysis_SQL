# 1. 사용자의 액션 수 집계하기
```sql
  WITH base AS (
      SELECT
          action,
          session, -- user_id 대신 session을 식별자로 사용
          COUNT(*) AS action_cnt
      FROM action_log
      GROUP BY action, session
  )
  SELECT
      action,
      -- 1. 해당 액션이 발생한 총 세션 수 (비회원 방문도 각각 카운트됨)
      COUNT(DISTINCT session) AS session_count, 
      
      -- 2. 세션당 해당 액션 평균 발생 횟수
      ROUND(AVG(action_cnt), 2) AS avg_count_per_session, 
      
      -- 3. 전체 세션 대비 해당 액션 발생 세션 비율 (Usage Rate)
      ROUND(100.0 * COUNT(DISTINCT session) / (SELECT COUNT(DISTINCT session) FROM action_log), 2) AS session_usage_rate
  FROM base
  GROUP BY action
  ORDER BY session_usage_rate DESC;
```
* 세션당 체류 액션 수 (Deep-dive)
  ```sql
    SELECT
      CASE WHEN user_id IS NULL THEN 'Guest' ELSE 'Member' END AS user_type,
      COUNT(DISTINCT session) AS total_sessions,
      COUNT(*) AS total_actions,
      -- 세션당 평균 액션 수 (방문 당 활동 밀도)
      ROUND(COUNT(*)::numeric / COUNT(DISTINCT session), 2) AS actions_per_session
  FROM action_log
  GROUP BY user_type;
  ```
# 2. 로그인/비로그인 사용자 구분

```sql
select

  ```

* 로그인 여부에 따른 행동 비중 분석
 ```sql
  select
  case when user_id is null then 'Guest' else ' Logged-in' end as user_group,
  action,
  count(*) as action_count
  from user_activities
  group by user_group,action
  order by user_group,action_count desc
  --- 비로그인 사용자가 주로 어떤 액션까지만 수행하고 멈추는지 확인 가능.
  ```
* 카테고리별 로그인 유저의 구매 전환(Purchase) 확인
```sql
  select
  category,
  count(*) filter (where user_id is null) as guest_actions,
  count(*) filter (where user_id is not null) as member_actions,
  count(*) filter (where action='purchase') as total_purchases
  from user_activities
  group by category
---`count(*) filter (where user_id is null)`: 해당 카테고리에서 **비로그인 사용자**가 발생시킨 모든 활동의 총합
--- `count(*) filter (where user_id is not null)`: 해당 카테고리에서 **로그인 사용자**가 발생시킨 모든 활동의 총합
--- `count(*) filter (where action='purchase')`: 해당 카테고리에서 로그인 여부와 상관없이 발생한 **실제 구매 건수**
```
* 회원/비회원 구매 분리 분석
```sql
select
category,
count(*) filter (where user_id is null and action='purchase') as guest_purchases,
count(*) filter (where user_id is not null and action='purchase') as member_purchases
from user_activities
group by category
```
* 구매 전환율(Conversion Rate)계산
```sql
select
category,
count(*) filter (where user_id is null and action='purchase') as guest_purchases,
count(*) filter (where user_id is not null and action='purchase') as member_purchases,
round(100.0*count(*) filter (where user_id is null and action='purchase')::numeric/
nullif(count(*) filter (where user_id is null),0),2) as guest_cvr,
round(100.0*count(*) filter (where user_id is not null and action='purchase')::numeric/
nullif(count(*) filter (where user_id is not null),0),2) as member_cvr
from user_activities
group by category
```

 ## 회원과 비회원을 구분해서 집계하기
 * 회원&비회원 집계
   ```sql
    with
    session_user_map as (
   --- 각 세션별로 "로그인한 유저가 있는지" 미리 정의
     select
    	session,
    	max(user_id) as fixed_user_id,
    	min(stamp) filter (where user_id is not null) as login_time
    from action_log
    group by session)
    
    select
      l.session,
      l.user_id,
      l.action,
      l.stamp,
      case
      when m.fixed_user_id is null then 'none' -- 세션 내내 로그인 안 함
      when l.stamp>=m.login_time then 'member' -- 로그인 시점 이후
      else 'guest_before_login' end as member_status --로그인 하기 전의 활
    from action_log as l
    left join session_user_map as m
    on l.session=m.session
   ```
* 로그인 전환 유도 목적 (Conversion Focus)
  ```sql
  with
  session_metrics as (
  select
  	session,
  	min(stamp) filter (where user_id is not null) as login_timestamp
  from action_log
  group by session)
  
  select
  l.session,
  l.action,
  l.products,
  case
  when m.login_timestamp is null then 'Pure_Guest' -- 끝까지 로그인 안 함
  when l.stamp<m.login_timestamp then 'Guest_Before_Login' --로그인 전 탐색 중
  when l.stamp=m.login_timestamp then 'Login_Moment' -- 로그인 직후 액션
  else 'Member_After_Login' end as behavioral_stage, --로그인 후 활동
  l.stamp
  from action_log as l
  left join session_metrics as m
  on l.session=m.session
  ```
* 순수 매출 집계
  ```sql
    with
  session_summary as (
  select
  	session,
  	max(user_id) as final_user_id,
  	sum(amount) as session_amount,
  	count(*) filter(where action='purchase') as purchase_count
  from action_log
  group by session)
  
  select
  case
  when final_user_id is not null then 'member_session'
  else 'guest_session' end as session_type,
  count(distinct session) as total_sessions,
  sum(session_amount) as total_revenue,
  round(sum(purchase_count)::numeric/nullif(count(distinct session),0),3)
  as cvr
  from session_summary
  group by session_type
  ```
  * CVR (Conversion Rate): `구매 횟수/총 세션 수`를 계산하여, 어떤 그룹이 더 효율적으로 구매에 도달하는지 보여준다.

## 3.연령별 구분 집계하기
* 성별과 연령으로 연령별 구분을 계산하는 쿼리
  ```sql
    with
    mst_users_with_int_birth_date as (
    select
    	*,
    	20260101 as int_specific_date,
    	cast(replace(birth_date,'-','') as integer) as int_birth_date
    	from mst_users
    ),
    
    mst_users_with_age as (
    select
    	user_id,
    	sex,
    	birth_date,
    	floor((int_specific_date-int_birth_date)/10000) as age
    from mst_users_with_int_birth_date),
    
    mst_users_with_category as (
    select
    	user_id,
    	sex,
    	age,
    	concat(
    	case when 20<=age then sex else '' end,
    	case 
    	when age between 4 and 12 then  'C'
    	when age between 13 and 19 then 'T'
    	when age between 20 and 34 then '1'
    	when age between 35 and 49 then '2'
    	when age>=50 then '3'
    	end) as category
    from mst_users_with_age)
    
    select * from mst_users_with_category
  ```
* 연령별 구분의 특징 추출하기
   ```sql
   with
    user_age_calc as (
    select
    	user_id,
    	sex,
    	date_part('year', age('2026-01-01'::date,birth_date::date)) as age
    from mst_users),
    
    user_segmentation as (
    select
    	*,
    	case
    	when age between 4 and 12 then 'C'
    	when age between 13 and 19 then 'T'
    	when age>=20 then
    	case
    	when age between 20 and 34 then concat(sex,'1')
    	when age between 35 and 39 then concat(sex,'2')
    	when age>=50 then concat(sex,'3')
    	end
    	else 'KIDS' end as category
    from user_age_calc)
    
    select
    l.category as product_category,
    u.category as user_category,
    count(*) as purchase_count
    from action_log as l
    join user_segmentation as u
    on l.user_id=u.user_id
    where l.action='purchase'
    group by l.category,u.category
    order by l.category,u.category
  ```

## 4.사용자의 방문 빈도 집계하기
* 한 주에 며칠 사용되었는지를 집계하는 쿼리
----
* 비회원용 식별자 컬럼(guest_id 또는 cookie_id)가 존재하는 경우
  ```sql
     WITH action_day_count_per_user AS (
      SELECT
          -- user_id가 없으면 guest_id를 사용해 비회원도 각각 고유하게 식별
          COALESCE(user_id, guest_id) AS distinct_user_id,
          COUNT(DISTINCT stamp::date) AS action_day_count
      FROM action_log
      WHERE stamp >= '2026-05-03 00:00:00' AND stamp < '2026-05-10 00:00:00'
      GROUP BY COALESCE(user_id, guest_id)
  )
  SELECT
      action_day_count,
      COUNT(*) AS user_count
  FROM action_day_count_per_user
  GROUP BY action_day_count
  ORDER BY action_day_count;
  ```
* session_id를 통해서 집계하는 경우
  ```sql
    with
  action_day as (
  select
  	session,
  	count(distinct stamp::date) as action_day_count
  from action_log
  WHERE stamp >= '2026-05-03 00:00:00' AND stamp < '2026-05-10 00:00:00'
  group by session)
  
  select
  action_day_count,
  count(*) as users
  from action_day
  group by action_day_count
  order by action_day_count  
  ```

* 구성비와 구성비누계를 계산하는 쿼리
  ```sql
    with
    action_day as (
    select
    	session,
    	count(distinct stamp::date) as action_day_count
    from action_log
    WHERE stamp >= '2026-05-03 00:00:00' AND stamp < '2026-05-10 00:00:00' 
    group by session)
    
  
  select
  action_day_count,
  count(*) as user_count,
  100.0*count(*)/sum(count(*)) over () as composition_ratio,
  100.0*sum(count(*)) over (order by action_day_count rows between unbounded preceding and current row)/
  sum(count(*)) over() as cumulative_ratio
  FROM action_day_count_per_user
  GROUP BY action_day_count
  ORDER BY action_day_count;
  ```
  * 방문 빈도별 매출 기여도 분석
    ```sql
    WITH action_day_count_per_user AS (
          SELECT
              session,
              COUNT(DISTINCT CAST(stamp AS DATE)) AS action_day_count,
              SUM(amount) AS total_amount -- 유저별 총 구매 금액
          FROM action_log
          WHERE stamp >= '2026-05-03 00:00:00' AND stamp < '2026-05-10 00:00:00'
          GROUP BY session
      )

      select
      action_day_count,
      count(*) as user_count,
      sum(total_amount) as total_revenue,
      round(avg(total_amount),2) as average_revenue,-- 인당 평균 매출
      round(100.0*sum(total_amount)/sum(sum(total_amount)) over(),2) as revenue_ratio
      from action_day_count_per_user
      group by action_day_count
      order by action_day_count
    ```
    * "매일 오는 사람보다 주 3회 오는 사람의 객단가가 더 높은가?"
    * "매출의 80%를 만드는 핵심 빈도 구간은 어디인가?"
* 카테고리별 방문 유도 기여도 분석
  ```sql
    with
  user_freq as (
  select
  	session,
  	count(distinct cast(stamp as date)) as total_freq
  from action_log
  WHERE stamp >= '2026-05-03 00:00:00' AND stamp < '2026-05-10 00:00:00'
  GROUP BY session)
  
  select
  a.category,
  round(avg(f.total_freq),2) as avg_visit_frequency
  from action_log as a
  join user_freq f 
  on a.session=f.session
  WHERE stamp >= '2026-05-03 00:00:00' AND stamp < '2026-05-10 00:00:00'
  group by a.category
  order by avg_visit_frequency desc
  ```
